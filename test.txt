type TreeMap struct {
	// 根据key值存储对应的树
	child  map[int]*Tree

	// 存储某个key值的节点当前被挂在的对应子树
	parent map[int]*Tree

	//
	minAlign int

	//
	maxAlign int
}

type Tree struct {
	// 父亲节点
	root int

	// 子节点
	childMap map[int]int

	// 子节点数据
	childData []int

	// 当前竖行 以0行为起始，不断减值
	alignLine int
}

func test(key int, path [][]int) {
	root := NewTree(len(path[0]) - 2)
	root.root = key
	rootChild := make([]int, 0, len(path[0])-2)
	for _, c := range path[1] {
		if c == 1 {
			rootChild = append(rootChild, c)
			// TreeMap.paren[c] = root
		}
	}
	//TreeMap.min = len(rootChild)
	root.childData = rootChild
	//
	// 判断是否有兄弟节点间存在相互引用
}

func NewTree(n int) *Tree {
	return &Tree{
		childMap:  make(map[int]int, n),
		childData: make([]int, 0, n),
		alignLine: 0,
	}
}
